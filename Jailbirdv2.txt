-- ================== HELPER: WALL CHECK ==================
local function isWallBetween(origin, targetPos, ignoreList)
    local direction = (targetPos - origin).Unit
    local ray = Ray.new(origin, direction * 1000)
    local hit = workspace:FindPartOnRayWithIgnoreList(ray, ignoreList or {LocalPlayer.Character})
    if hit and not hit:IsDescendantOf(LocalPlayer.Character) then
        if not hit.Parent:FindFirstChildOfClass("Humanoid") then
            return true
        end
    end
    return false
end

-- ================== AIMBOT ==================
local smoothActive, stickyActive = false, false
local function getClosestInFOV()
    local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    local best, bestDist = nil, aimbot.FOV
    local camPos = Camera.CFrame.Position
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character then
            local head = plr.Character:FindFirstChild("Head")
            local humanoid = plr.Character:FindFirstChildOfClass("Humanoid")
            if not head or not humanoid or humanoid.Health <= 0 then continue end
            if aimbot.TeamCheck and plr.Team == LocalPlayer.Team then continue end
            local targetPos = head.Position
            local distToTarget = (targetPos - camPos).Magnitude
            if distToTarget > aimbot.Distance then continue end
            if aimbot.WallCheck and isWallBetween(camPos, targetPos, {LocalPlayer.Character, plr.Character}) then continue end
            local screenPos, onScreen = Camera:WorldToViewportPoint(targetPos)
            if onScreen then
                local dist = (Vector2.new(screenPos.X, screenPos.Y) - center).Magnitude
                if dist < bestDist then best, bestDist = plr, dist end
            end
        end
    end
    return best
end

addConn(RunService.RenderStepped:Connect(function()
    if aimbot.Enabled then
        local target = getClosestInFOV()
        if target and target.Character then
            local root = target.Character:FindFirstChild("HumanoidRootPart")
            local head = target.Character:FindFirstChild("Head")
            if stickyActive and head then
                Camera.CFrame = CFrame.new(Camera.CFrame.Position, head.Position)
            elseif smoothActive and root then
                Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, root.Position), aimbot.Smoothness)
            end
        end
    end
    if aimbot.ShowFOV and aimbot.Enabled and hasDrawing then
        fovCircle.Visible = true; fovCircle.Radius = aimbot.FOV
        fovCircle.Position = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)

        fovCircle.Color = col
    elseif hasDrawing then
        fovCircle.Visible = false
    end
end))

-- ================== WEATHER (ParticleEmitter) ==================
local function getVisualColor()
    local col = colorMap[visual.GlobalColor]
    if visual.GlobalColor == "Rainbow" then col = Color3.fromHSV(tick() % 6 / 6, 1, 1) end
    return col
end

local function updateWeather()
    if weatherEmitterPart then weatherEmitterPart:Destroy(); weatherEmitterPart = nil end
    if weatherParticlesFolder then weatherParticlesFolder:Destroy(); weatherParticlesFolder = nil end

    if visual.WeatherType == "Off" then return end

    weatherParticlesFolder = Instance.new("Folder")
    weatherParticlesFolder.Name = "BO2_Weather"
    weatherParticlesFolder.Parent = workspace

    weatherEmitterPart = Instance.new("Part")
    weatherEmitterPart.Name = "WeatherEmitter"
    weatherEmitterPart.Size = Vector3.new(1, 1, 1)
    weatherEmitterPart.Anchored = true
    weatherEmitterPart.CanCollide = false
    weatherEmitterPart.Transparency = 1
    weatherEmitterPart.Position = Vector3.new(0, 50, 0)
    weatherEmitterPart.Parent = weatherParticlesFolder

    local emitter = Instance.new("ParticleEmitter")
    emitter.Name = "WeatherEmitter"
    emitter.Rate = 100
    emitter.Lifetime = NumberRange.new(3, 5)
    emitter.Speed = NumberRange.new(20, 30)
    emitter.VelocityInheritance = 0
    emitter.Enabled = true
    emitter.Parent = weatherEmitterPart

    if visual.WeatherType == "Rain" then
        emitter.Texture = "rbxassetid://2703041744"
        emitter.Color = ColorSequence.new(getVisualColor())
    elseif visual.WeatherType == "Snow" then
        emitter.Texture = "rbxassetid://964974685"
        emitter.Color = ColorSequence.new(Color3.new(1,1,1))
        emitter.Speed = NumberRange.new(1, 3)
        emitter.Rate = 50
    elseif visual.WeatherType == "Money" then
        emitter.Texture = "rbxassetid://243573432"
        emitter.Color = ColorSequence.new(Color3.fromRGB(76,175,80))
        emitter.Size = NumberSequence.new(2)
    elseif visual.WeatherType == "Cash" then
        emitter.Texture = "rbxassetid://243573432"
        emitter.Color = ColorSequence.new(Color3.fromRGB(0,170,0))
        emitter.Size = NumberSequence.new(2)
    end
end

-- ================== PLAYER HAT (physical cylinders) ==================
local function createHatModel(targetPlayer)
    local model = Instance.new("Model")
    model.Name = "BO2_Hat_"..targetPlayer.Name
    local lines = {}
    for i=1,6 do
        local part = Instance.new("Part")
        part.Name = "Line"..i
        part.Size = Vector3.new(0.1, 1.5, 0.1)
        part.Anchored = true
        part.CanCollide = false
        part.BrickColor = BrickColor.new(getVisualColor())
        part.Material = Enum.Material.SmoothPlastic
        part.Parent = model
        table.insert(lines, part)
    end
    model.Parent = workspace
    return {model = model, lines = lines}
end

local function updateHat(plr, hatData)
    local char = plr.Character
    if not char or not char:FindFirstChild("Head") then return end
    local head = char.Head
    local top = head.Position + Vector3.new(0, 1.5, 0)
    local col = getVisualColor()
    for i, linePart in ipairs(hatData.lines) do
        local angle = (i-1) * (math.pi*2/6)
        local bottom = head.Position + Vector3.new(math.cos(angle)*0.8, 0.1, math.sin(angle)*0.8)
        local length = (top - bottom).Magnitude
        local mid = (bottom + top)/2
        linePart.Size = Vector3.new(0.1, length, 0.1)
        linePart.CFrame = CFrame.new(mid, top) * CFrame.Angles(math.pi/2, 0, 0)
        linePart.BrickColor = BrickColor.new(col)
    end
end

addConn(RunService.RenderStepped:Connect(function()
    for plr, hatData in pairs(hatModels) do
        if not plr.Parent or not plr.Character then
            hatData.model:Destroy()
            hatModels[plr] = nil
        end
    end

    if not visual.PlayerHat then
        for _, hatData in pairs(hatModels) do
            for _, part in ipairs(hatData.lines) do part.Transparency = 1 end
        end
        return
    end

    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("Head") then
            local humanoid = plr.Character:FindFirstChildOfClass("Humanoid")
            if humanoid and humanoid.Health > 0 then
                local hatData = hatModels[plr]
                if not hatData then
                    hatData = createHatModel(plr)
                    hatModels[plr] = hatData
                end
                updateHat(plr, hatData)
            else
                if hatModels[plr] then
                    hatModels[plr].model:Destroy()
                    hatModels[plr] = nil
                end
            end
        end
    end
end))

local function createESP(player) return {
    Box = newDrawing("Square"), HealthBar = newDrawing("Square"),
    Name = newDrawing("Text"), Distance = newDrawing("Text"), Line = newDrawing("Line"),
    Arrow = newDrawing("Line"),
} end
local function createSkeleton() local lines = {}; for _=1,12 do table.insert(lines, newDrawing("Line")) end; return lines end

local function getEdgePoint(screenPos)
    local center = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
    local dir = (Vector2.new(screenPos.X, screenPos.Y) - center).Unit
    local screenMin = Vector2.new(0,0)
    local screenMax = Vector2.new(Camera.ViewportSize.X, Camera.ViewportSize.Y)
    local tx1 = (screenMin.X - center.X) / dir.X
    local tx2 = (screenMax.X - center.X) / dir.X
    local ty1 = (screenMin.Y - center.Y) / dir.Y
    local ty2 = (screenMax.Y - center.Y) / dir.Y
    local tValues = {}
    if tx1 > 0 then table.insert(tValues, tx1) end
    if tx2 > 0 then table.insert(tValues, tx2) end
    if ty1 > 0 then table.insert(tValues, ty1) end
    if ty2 > 0 then table.insert(tValues, ty2) end
    table.sort(tValues)
    local t = tValues[1] or 0
    return center + dir * t
end

addConn(RunService.RenderStepped:Connect(function()
    for plr, drawings in pairs(espObjects) do
        local char = plr.Character
        local head = char and char:FindFirstChild("Head")
        local humanoid = char and char:FindFirstChildOfClass("Humanoid")
        local alive = humanoid and humanoid.Health > 0
        local show = true
        if not plr.Parent or not head then show = false
        elseif esp.DeadCheck and not alive then show = false
        elseif esp.TeamCheck and plr.Team == LocalPlayer.Team then show = false
        end
        if esp.WallCheck and show and head then
            if isWallBetween(Camera.CFrame.Position, head.Position, {LocalPlayer.Character, plr.Character}) then
                show = false
            end
        end

        if not show or not esp.Enabled then
            for _, v in pairs(drawings) do v.Visible = false end
            if skeletonLines[plr] then for _, v in ipairs(skeletonLines[plr]) do v.Visible = false end end
            continue
        end

        local root = char:FindFirstChild("HumanoidRootPart")
        local headPos, onScreen = Camera:WorldToViewportPoint(head.Position)
        if onScreen then
            local bottom = root and Camera:WorldToViewportPoint(root.Position - Vector3.new(0,3,0)) or nil
            local top = Camera:WorldToViewportPoint(head.Position + Vector3.new(0,0.5,0))
            if bottom and top then
                local boxH = bottom.Y - top.Y; local boxW = boxH * 0.35; local boxX = top.X - boxW/2; local boxY = top.Y
                if esp.Boxes then
                    drawings.Box.Visible = true; drawings.Box.Size = Vector2.new(boxW, boxH)
                    drawings.Box.Position = Vector2.new(boxX, boxY); drawings.Box.Thickness = 1; drawings.Box.Filled = false
                    drawings.Box.Color = esp.BoxColor == "Rainbow" and Color3.fromHSV(tick()%6/6,1,1) or colorMap[esp.BoxColor] or Color3.new(1,1,1)
                else drawings.Box.Visible = false end
                if esp.HealthBar then
                    local health = humanoid.Health / humanoid.MaxHealth
                    drawings.HealthBar.Visible = true; drawings.HealthBar.Size = Vector2.new(2, boxH * health)
                    drawings.HealthBar.Position = Vector2.new(boxX-4, boxY + boxH*(1-health))
                    drawings.HealthBar.Color = Color3.fromRGB((1-health)*255, health*255, 0); drawings.HealthBar.Filled = true
                else drawings.HealthBar.Visible = false end
                if esp.Names then
                    drawings.Name.Visible = true; drawings.Name.Text = plr.Name; drawings.Name.Size = 14
                    drawings.Name.Position = Vector2.new(top.X, top.Y-15); drawings.Name.Center = true
                    drawings.Name.Color = Color3.new(1,1,1)
                else drawings.Name.Visible = false end
                if esp.Distance then
                    local dist = (Camera.CFrame.Position - head.Position).Magnitude
                    drawings.Distance.Visible = true; drawings.Distance.Text = string.format("%.0f", dist); drawings.Distance.Size = 14
                    drawings.Distance.Position = Vector2.new(top.X, bottom.Y+5); drawings.Distance.Center = true
                    drawings.Distance.Color = Color3.new(1,1,1)
                else drawings.Distance.Visible = false end
                if esp.Lines then
                    drawings.Line.Visible = true; drawings.Line.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y)
                    drawings.Line.To = Vector2.new(bottom.X, bottom.Y)
                    drawings.Line.Color = esp.LineColor == "Rainbow" and Color3.fromHSV(tick()%6/6,1,1) or colorMap[esp.LineColor] or Color3.new(1,1,1)
                else drawings.Line.Visible = false end
                if esp.Arrows then
                    drawings.Arrow.Visible = true
                    drawings.Arrow.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y/2)
                    drawings.Arrow.To = Vector2.new(headPos.X, headPos.Y)
                    drawings.Arrow.Color = colorMap[esp.ArrowColor] or Color3.new(1,1,1)
                else drawings.Arrow.Visible = false end
            end
        else
            if esp.Arrows then
                local edgePoint = getEdgePoint(headPos)
                drawings.Arrow.Visible = true
                drawings.Arrow.From = edgePoint
                drawings.Arrow.To = Vector2.new(headPos.X, headPos.Y)
                drawings.Arrow.Color = colorMap[esp.ArrowColor] or Color3.new(1,1,1)
            else drawings.Arrow.Visible = false end
            for _, v in pairs({drawings.Box, drawings.HealthBar, drawings.Name, drawings.Distance, drawings.Line}) do v.Visible = false end
        end
    end

    -- Chams
    if esp.Enabled and esp.Chams then
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                local humanoid = plr.Character:FindFirstChildOfClass("Humanoid")
                if humanoid and humanoid.Health > 0 and (not esp.TeamCheck or plr.Team ~= LocalPlayer.Team) then
                    local col = esp.ChamColor == "Rainbow" and Color3.fromHSV(tick()%6/6,1,1) or colorMap[esp.ChamColor] or Color3.new(1,1,1)
                    for _, part in ipairs(plr.Character:GetChildren()) do if part:IsA("BasePart") then part.Color = col end end
                end
            end
        end
    end

    -- Skeleton
    if esp.Enabled and esp.Skeleton then
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character then
                local humanoid = plr.Character:FindFirstChildOfClass("Humanoid")
                if humanoid and humanoid.Health > 0 and (not esp.TeamCheck or plr.Team ~= LocalPlayer.Team) then
                    local lines = skeletonLines[plr] or createSkeleton()
                    skeletonLines[plr] = lines
                    local connections = {
                        {"Head","UpperTorso"}, {"UpperTorso","LowerTorso"},
                        {"UpperTorso","LeftUpperArm"}, {"LeftUpperArm","LeftLowerArm"}, {"LeftLowerArm","LeftHand"},
                        {"UpperTorso","RightUpperArm"}, {"RightUpperArm","RightLowerArm"}, {"RightLowerArm","RightHand"},
                        {"LowerTorso","LeftUpperLeg"}, {"LeftUpperLeg","LeftLowerLeg"}, {"LeftLowerLeg","LeftFoot"},
                        {"LowerTorso","RightUpperLeg"}, {"RightUpperLeg","RightLowerLeg"}, {"RightLowerLeg","RightFoot"},
                    }
                    for i, conn in ipairs(connections) do
                        local a = plr.Character:FindFirstChild(conn[1]); local b = plr.Character:FindFirstChild(conn[2])
                        if a and b and lines[i] then
                            lines[i].Visible = true; lines[i].From = Camera:WorldToViewportPoint(a.Position)
                            lines[i].To = Camera:WorldToViewportPoint(b.Position)
                            lines[i].Color = esp.SkeletonColor == "Rainbow" and Color3.fromHSV(tick()%6/6,1,1) or colorMap[esp.SkeletonColor] or Color3.new(1,1,1)
                        elseif lines[i] then lines[i].Visible = false end
                    end
                end
            end
        end
    else
        for _, lines in pairs(skeletonLines) do for _, v in ipairs(lines) do v.Visible = false end end
    end
end))

for _, plr in ipairs(Players:GetPlayers()) do if plr ~= LocalPlayer then espObjects[plr] = createESP(plr) end end

function refresh()
    for _, lbl in ipairs(optionLabels) do lbl:Destroy() end
    optionLabels = {}
    local opts = menus[currentMenu]
    if not opts then return end
    for i, opt in ipairs(opts) do
        local row = Instance.new("Frame")
        row.Size = UDim2.new(1,0,0,28); row.Position = UDim2.new(0,0,0,(i-1)*28)
        row.BackgroundTransparency = 1; row.Parent = menuArea

        if i == selectedIndex then
            local sel = Instance.new("Frame")
            sel.Size = UDim2.new(1,0,1,0); sel.BackgroundColor3 = Color3.fromRGB(108,155,255)
            sel.BorderSizePixel = 0; sel.BackgroundTransparency = 0.3; sel.Parent = row
        end

        local arrow = Instance.new("TextLabel")
        arrow.Size = UDim2.new(0,16,1,0); arrow.Position = UDim2.new(0,4,0,0)
        arrow.BackgroundTransparency = 1; arrow.Text = (i == selectedIndex) and ">" or ""
        arrow.Font = Enum.Font.GothamBold; arrow.TextColor3 = Color3.new(1,1,1); arrow.TextSize = 20
        arrow.Parent = row

        local label = Instance.new("TextLabel")
        label.Size = UDim2.new(0,180,1,0); label.Position = UDim2.new(0,22,0,0)
        label.BackgroundTransparency = 1
        local displayText = opt.text
        if opt.dynamicText then
            displayText = opt.dynamicText()
        elseif opt.value ~= nil then
            if opt.options then
                displayText = opt.text .. ": " .. tostring(opt.value)
            else
                displayText = opt.text .. ": " .. tostring(opt.value)
            end
        end
        label.Text = displayText
        label.Font = Enum.Font.GothamBold; label.TextColor3 = Color3.new(1,1,1); label.TextSize = 16
        label.TextXAlignment = Enum.TextXAlignment.Left
        label.Parent = row

        if opt.sub then
            local subInd = Instance.new("TextLabel")
            subInd.Size = UDim2.new(0,20,1,0); subInd.Position = UDim2.new(1,-22,0,0)
            subInd.BackgroundTransparency = 1; subInd.Text = ">>"
            subInd.Font = Enum.Font.GothamBold; subInd.TextColor3 = Color3.fromRGB(122,0,0); subInd.TextSize = 18
            subInd.Parent = row
        end

        table.insert(optionLabels, row)
    end
    menuArea.CanvasSize = UDim2.new(0,0,0, #opts * 28)
end

function goBack()
    if sliderVisible then exitSlider() end
    if #menuStack > 1 then
        table.remove(menuStack)
        currentMenu = menuStack[#menuStack]
        selectedIndex = 1
        refresh()
    end
end

-- ================== SLIDER SYSTEM ==================
local sliderVisible = false
local sliderOpt = nil
local sliderVal = 0
local sliderMinN = 0
local sliderMaxN = 100
local sliderStepN = 1
local sliderIsOptions = false

function updateSliderBar()
    if not sliderVisible then return end
    local fraction = (sliderVal - sliderMinN) / (sliderMaxN - sliderMinN)
    sliderFill.Size = UDim2.new(fraction, 0, 1, 0)
    sliderThumb.Position = UDim2.new(fraction, -4, 0.5, -7)
    local displayValue = sliderIsOptions and sliderOpt.options[sliderVal] or sliderVal
    if sliderIsOptions then
        sliderLabel.Text = sliderOpt.text .. ": " .. tostring(displayValue)
        sliderMinLabel.Text = ""
        sliderMaxLabel.Text = ""
    else
        sliderLabel.Text = sliderOpt.text .. ": " .. string.format(sliderStepN < 1 and "%.2f" or "%.0f", sliderVal)
        sliderMinLabel.Text = tostring(sliderMinN)
        sliderMaxLabel.Text = tostring(sliderMaxN)
    end
end

function enterSlider(opt)
    sliderVisible = true
    sliderOpt = opt
    if opt.options then
        sliderIsOptions = true
        local idx = table.find(opt.options, opt.value) or 1
        sliderVal = idx
        sliderMinN = 1
        sliderMaxN = #opt.options
        sliderStepN = 1
    else
        sliderIsOptions = false
        sliderVal = opt.value
        sliderMinN = opt.min
        sliderMaxN = opt.max
        sliderStepN = opt.step
    end
    menuArea.Visible = false
    sliderFrame.Visible = true
    updateSliderBar()
end

function exitSlider()
    sliderVisible = false
    sliderFrame.Visible = false
    menuArea.Visible = true
    if sliderOpt then
        if sliderIsOptions then
            sliderOpt.setter(sliderOpt.options[sliderVal])
        else
            sliderOpt.setter(sliderVal)
        end
        sliderOpt = nil
    end
    refresh()
end

function adjustSlider(delta)
    if not sliderVisible then return end
    sliderVal = sliderVal + delta * sliderStepN
    if sliderVal < sliderMinN then sliderVal = sliderMinN end
    if sliderVal > sliderMaxN then sliderVal = sliderMaxN end
    updateSliderBar()
end

-- ================== RELIABLE TOGGLE (DPadLeft + M key) ==================
local menuOpen = false
local dPadWasDown = false

addConn(RunService.RenderStepped:Connect(function()
    local isDown = false
    pcall(function()
        isDown = UIS:IsGamepadButtonDown(Enum.KeyCode.DPadLeft)
    end)
    if isDown and not dPadWasDown then
        menuOpen = not menuOpen
        mainFrame.Visible = menuOpen
        if menuOpen then
            currentMenu = "main"
            selectedIndex = 1
            menuStack = {"main"}
            refresh()
        end
    end
    dPadWasDown = isDown
end))

addConn(UIS.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.M then
        menuOpen = not menuOpen
        mainFrame.Visible = menuOpen
        if menuOpen then
            currentMenu = "main"
            selectedIndex = 1
            menuStack = {"main"}
            refresh()
        end
    end
end))

-- ================== NAVIGATION ==================
addConn(UIS.InputBegan:Connect(function(input, _)
    if not mainFrame.Visible then return end
    if input.UserInputType == Enum.UserInputType.Gamepad1 then
        if sliderVisible then
            if input.KeyCode == Enum.KeyCode.DPadRight then adjustSlider(1)
            elseif input.KeyCode == Enum.KeyCode.DPadDown then exitSlider() end
            -- DPadLeft handled in RenderStepped toggle (avoid conflict)
        else
            local opts = menus[currentMenu]
            if not opts then return end
            if input.KeyCode == Enum.KeyCode.ButtonR1 then
                selectedIndex = selectedIndex - 1; if selectedIndex < 1 then selectedIndex = #opts end; refresh()
            elseif input.KeyCode == Enum.KeyCode.ButtonL1 then
                selectedIndex = selectedIndex + 1; if selectedIndex > #opts then selectedIndex = 1 end; refresh()
            elseif input.KeyCode == Enum.KeyCode.DPadDown then
                local sel = opts[selectedIndex]
                if sel then
                    if sel.sub then
                        table.insert(menuStack, sel.sub)
                        currentMenu = sel.sub
                        selectedIndex = 1
                        refresh()
                    elseif sel.action then
                        sel.action()
                    elseif sel.value ~= nil then
                        enterSlider(sel)
                    end
                end
            end
        end
    end
end))
